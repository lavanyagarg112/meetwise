# One-Time Use Commands
```
CREATE TABLE Organisations(
ID INTEGER PRIMARY KEY AUTOINCREMENT,
Name TEXT NOT NULL UNIQUE,
Owner INT NOT NULL,
Visibility TEXT NOT NULL DEFAULT "PRIVATE",
FOREIGN KEY(Owner) REFERENCES Users(ID)
)
```

```
 CREATE TABLE USERS(
 ID INTEGER PRIMARY KEY AUTOINCREMENT,
 USERNAME TEXT UNIQUE NOT NULL,
 EMAIL TEXT UNIQUE NOT NULL, 
 PASSWORD TEXT NOT NULL,
 FIRSTNAME TEXT NOT NULL,
 LASTNAME TEXT NOT NULL,
 ACTIVEORG INT,
 VISIBILITY TEXT NOT NULL DEFAULT `PRIVATE`,
 FOREIGN KEY (ACTIVEORG) REFERENCES Organisations(ID)
 )  
```

```
CREATE  TABLE UserOrg(
ID INTEGER NOT NULL,
ORGANISATION INTEGER NOT NULL,
FOREIGN KEY (ID) REFERENCES Users(ID),
FOREIGN KEY (ORGANISATION) REFERENCES Organisations(ID)
)     
```

```
CREATE  TABLE PendingInvites(
EMAIL TEXT NOT NULL,
ORGANISATION INT NOT NULL,
ROLE TEXT NOT NULL,
FOREIGN KEY (ORGANISATION) REFERENCES Organisations(ID),
PRIMARY KEY (EMAIL, ORGANISATION)
)
```
